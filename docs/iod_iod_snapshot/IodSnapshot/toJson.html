Implementation

Map<String, dynamic> toJson() => {
      "millisecondsSinceEpoch": createdMillisecondsSinceEpoch,
      "light": light,
      "proximitySensorTriggered": proximitySensorTriggered,
      "magneticField": {
        "x": magneticField.x,
        "y": magneticField.y,
        "z": magneticField.z,
      },
      "acceleration": {
        "x": acceleration.x,
        "y": acceleration.y,
        "z": acceleration.z,
      },
      "userAcceleration": {
        "x": userAcceleration.x,
        "y": userAcceleration.y,
        "z": userAcceleration.z,
      },
      "gyroscope": {
        "x": gyroscope.x,
        "y": gyroscope.y,
        "z": gyroscope.z,
      },
      "rotation": {
        "yaw": rotation.yaw,
        "pitch": rotation.pitch,
        "roll": rotation.roll,
      },
      "orientation": {
        "yaw": orientation.yaw,
        "pitch": orientation.pitch,
        "roll": orientation.roll,
      },
      "location": {
        "lat": position.latitude,
        "lng": position.longitude,
      },
      "weather": {
        "cloudiness": weather.cloudiness,
        "temperature": weather.temperature.kelvin,
      },
      "routers": List<int>.from(networks.map((network) => network.level)),
      "cells": List<int>.from(cellSignalStrengths.where((x) => x.dbm != -113).map((x) => x.dbm)),
      "sattelites": List<double>.from(gnss.measurements.map((measurement) => measurement.cn0DbHz)),
    };
