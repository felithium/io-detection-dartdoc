Future<void> init()

Initialize all streams to update the current state if an event appears

Implementation

static Future<void> init() async {
  gnss = GnssMeasurementModel(measurements: []);
  lightSensorStream.listen((event) => light = event);
  proximityStream.listen((event) => proximitySensorTriggered = event.getValue());
  magneticFieldStream.listen((event) => magneticField = event);
  accelerometerStream.listen((event) => acceleration = event);
  linearAccelerationStream.listen((event) => userAcceleration = event);
  gyroscopeStream.listen((event) => gyroscope = event);
  orientationStream.listen((event) => rotation = event);
  rotationVectorStream.listen((event) => orientation = event);
  geolocationStream.listen((event) {
    if (position != null && position.isMocked) throw Error();
    position = event;
  });
  weatherStream.listen((event) => weather = event);
  wifiStream.listen((event) => networks = event);
  cellStream.listen((event) => cellSignalStrengths = event);
  gnssStream.listen((event) => gnss = event);
  await Future.wait([
    accelerometerStream.first.then((event) => acceleration = event),
    gyroscopeStream.first.then((event) => gyroscope = event),
    lightSensorStream.first.then((event) => light = event),
    linearAccelerationStream.first.then((event) => userAcceleration = event),
    magneticFieldStream.first.then((event) => magneticField = event),
    orientationStream.first.then((event) => rotation = event),
    proximityStream.first.then((event) => proximitySensorTriggered = event.getValue()),
    rotationVectorStream.first.then((event) => orientation = event),
    geolocationStream.first.then((event) => position = event),
    cellStream.first.then((event) => cellSignalStrengths = event),
    // gnssStream.first,
    wifiStream.first.then((event) => networks = event),
    weatherStream.first.then((event) => weather = event),
  ]);
}
