Future<List<LatLng>> readThmEntrances()

Helper to read entrances from the buildings of the THM Uses the indoor-data-service

Implementation

Future<List<LatLng>> readThmEntrances() async {
  var response = await http.get(Uri.parse("https://mocainfo.thm.de/indoor-model/api/v2/buildings"));
  var body = jsonDecode(response.body);
  var buildings = body["buildings"]["content"];
  List<LatLng> thmDoors = [];
  for (var building in buildings) {
    var doors = building["levels"][0]["doors"];
    for (var door in doors) {
      if (door.containsKey("entrance")) {
        var point = door["point"];
        thmDoors.add(LatLng(point["lat"], point["lng"]));
      }
    }
  }
  return thmDoors;
}
