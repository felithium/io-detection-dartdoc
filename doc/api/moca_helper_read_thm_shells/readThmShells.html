Future<List<List<LatLng>>> readThmShells()

Helper to read shells from the buildings of the THM Uses the indoor-data-service

Implementation

Future<List<List<LatLng>>> readThmShells() async {
  var response = await http.get(Uri.parse("https://mocainfo.thm.de/indoor-model/api/v2/buildings"));
  var body = jsonDecode(response.body);
  var buildings = body["buildings"]["content"];
  List<List<LatLng>> result = [];
  for (int i = 0; i < buildings.length; i++) {
    var rooms = buildings[i]["levels"][0]["rooms"];
    for (int j = 0; j < rooms.length; j++) {
      var room = rooms[j];
      var points = room["points"];
      result.add(points.map<LatLng>((point) => LatLng(point["lat"], point["lng"])).toList());
    }
  }
  return result;
}
